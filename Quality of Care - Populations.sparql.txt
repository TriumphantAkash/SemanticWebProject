fPREFIX DS1208: <http://logd.tw.rpi.edu/source/data-gov/dataset/1208/vocab/raw/>

SELECT distinct ?state (SUM(xsd:double(?mental_health_outpatient)) as ?Total_mental_health_outpatient) (SUM(xsd:double(?non_mental_health_outpatients)) as ?Total_non_mental_health_outpatient)
WHERE {  
GRAPH <http://logd.tw.rpi.edu/source/data-gov/dataset/1208/version/1st-anniversary>  {	
?s DS1208:mental_health_outpatient ?mental_health_outpatient.
?s DS1208:non_mental_health_outpatient ?non_mental_health_outpatient.
?s DS1208:state ?state .    
}
}
order by ?state
